循环遍历 + a 标签 href 改变
如何在a标签的href中传入需要的参数?

<ul class="dropdown-menu" id="app">
    <li v-for="date_list in date_list">
        <a :href = "get_date_Href(date_list)" >{{ date_list }}</a>
    </li>
</ul>

new Vue({
    el: '#app',
    data: {
        date_list: date_list
    },
    methods:{
        get_date_Href:function(val){
            return '/Phrase_Controller/index?Date='+val
        }
    }
})
